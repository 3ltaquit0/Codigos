            public static  void shellSort(ArrayList list, Comparator comparator) { // Método shellSort genérico
                int n = list.size(); // Obtiene el tamaño de la lista
                for (int gap = n / 2; gap > 0; gap /= 2) { // Inicializa el gap y lo reduce a la mitad en cada iteración
                    for (int i = gap; i < n; i++) { // Itera desde gap hasta el final de la lista
                        T temp = list.get(i); // Guarda el elemento actual en temp
                        int j;
                        for (j = i; j >= gap && comparator.compare(list.get(j - gap), temp) > 0; j -= gap) { // Compara y desplaza elementos
                            list.set(j, list.get(j - gap)); // Desplaza el elemento hacia adelante
                        }
                        list.set(j, temp); // Coloca temp en su posición correcta
                    }
                }
            }
        }
