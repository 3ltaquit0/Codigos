Explicación del Código de Serpientes y Escaleras

Este programa en Java implementa el juego clásico de "Serpientes y Escaleras". Los jugadores se desplazan en un tablero de 100 posiciones, donde algunas casillas contienen serpientes (que retroceden al jugador) o escaleras (que avanzan al jugador). El primer jugador en llegar a la posición 100 gana la partida.

Descripción de las Clases

La clase principal SerpientesEscaleras contiene dos clases internas: Nodo y Jugador.

  • Nodo: Representa una posición en el tablero, con un destino que apunta a otra posición si hay una serpiente o escalera.
  • Jugador: Representa a cada jugador y su posición actual en el tablero.

Funcionamiento Principal

El juego se desarrolla lanzando un dado que determina el avance en el tablero. Si el jugador cae en una casilla con serpiente o escalera, se desplaza a la posición correspondiente. 


import java.util.LinkedList;
import java.util.Random;
import java.util.Scanner;

public class SerpientesEscaleras {

    // Clase interna que representa cada casilla del tablero
    class Nodo {
        int posicion;          // Posición actual en el tablero
        Nodo siguiente;        // Nodo siguiente en el tablero
        int destino;           // Si es una serpiente/escalera, apunta al destino final

        Nodo(int posicion) {
            this.posicion = posicion;
            this.siguiente = null;
            this.destino = posicion; // Por defecto, la posición destino es la misma
        }
    }

    // Clase interna que representa a un jugador
    class Jugador {
        String nombre;        // Nombre del jugador
        Nodo posicionActual;  // Posición actual en el tablero

        Jugador(String nombre, Nodo inicio) {
            this.nombre = nombre;
            this.posicionActual = inicio; // El jugador empieza en el nodo inicial
        }
    }

    Nodo inicioTablero;        // Inicio del tablero
    LinkedList<Jugador> jugadores; // Lista de jugadores
    Random dado;               // Generador de números aleatorios (dado)
    final int META = 100;      // Meta del juego (posición 100)

    public SerpientesEscaleras() {
        jugadores = new LinkedList<>();
        dado = new Random();
        construirTablero();    // Inicializa el tablero con serpientes y escaleras
    }

    // Construcción del tablero con 100 posiciones
    private void construirTablero() {
        Nodo actual = new Nodo(1); // Nodo inicial
        inicioTablero = actual;

        for (int i = 2; i <= META; i++) { // Crea 100 nodos en el tablero
            Nodo nuevo = new Nodo(i);
            actual.siguiente = nuevo;
            actual = nuevo;
        }

        // Configuración de serpientes y escaleras en el tablero
        asignarSerpientesEscaleras(16, 6);  // Serpiente
        asignarSerpientesEscaleras(47, 26); // Serpiente
        asignarSerpientesEscaleras(49, 11); // Serpiente
        asignarSerpientesEscaleras(62, 19); // Serpiente
        asignarSerpientesEscaleras(1, 38);  // Escalera
        asignarSerpientesEscaleras(4, 14);  // Escalera
        asignarSerpientesEscaleras(9, 31);  // Escalera
        asignarSerpientesEscaleras(28, 84); // Escalera
    }

    // Asigna serpientes o escaleras en posiciones específicas
    private void asignarSerpientesEscaleras(int origen, int destino) {
        Nodo actual = inicioTablero;
        while (actual != null) {
            if (actual.posicion == origen) {
                actual.destino = destino; // Cambia el destino si es serpiente/escalera
                break;
            }
            actual = actual.siguiente;
        }
    }

    // Método para agregar jugadores al juego
    public void agregarJugadores(int cantidad) {
        try (Scanner scanner = new Scanner(System.in)) {
            for (int i = 1; i <= cantidad; i++) {
                System.out.print("Ingrese el nombre del jugador " + i + ": ");
                String nombre = scanner.nextLine();
                if (nombre != null && !nombre.trim().isEmpty()) {
                    jugadores.add(new Jugador(nombre, inicioTablero)); // Empieza en el nodo inicial
                } else {
                    System.out.println("Nombre inválido.");
                    i--; // Reintenta si el nombre no es válido
                }
            }
        }
    }

    // Método principal que inicia el juego
    public void iniciarJuego() {
        try (Scanner scanner = new Scanner(System.in)) {
            System.out.print("¿Cuántos jugadores (2-10)? ");
            int cantidad = scanner.nextInt();
            if (cantidad < 2 || cantidad > 10) {
                System.out.println("Número de jugadores inválido.");
                return;
            }

            agregarJugadores(cantidad); // Agrega los jugadores
            System.out.println("¡El juego comienza!");

            boolean juegoTerminado = false;

            while (!juegoTerminado) {
                for (Jugador jugador : jugadores) {
                    System.out.println("\nTurno de " + jugador.nombre);
                    int avance = lanzarDado();
                    System.out.println(jugador.nombre + " sacó un " + avance);

                    // Mueve al jugador en el tablero
                    jugador.posicionActual = moverJugador(jugador.posicionActual, avance);

                    // Verifica si el jugador ha ganado
                    if (jugador.posicionActual.posicion >= META) {
                        System.out.println("¡" + jugador.nombre + " ha ganado!");
                        juegoTerminado = true;
                        break;
                    } else {
                        System.out.println(jugador.nombre + " está en la posición " + jugador.posicionActual.posicion);
                    }
                }
            }
        }
    }

    // Mueve al jugador una cantidad de pasos en el tablero
    private Nodo moverJugador(Nodo posicionActual, int pasos) {
        Nodo actual = posicionActual;

        for (int i = 0; i < pasos && actual.siguiente != null; i++) {
            actual = actual.siguiente;
        }

        if (actual.destino != actual.posicion) {
            System.out.println("¡" + (actual.destino > actual.posicion ? "Escalera" : "Serpiente") + "! " + 
                               "Avanza a la posición " + actual.destino);
            actual = obtenerNodoPorPosicion(actual.destino); // Mueve al destino
        }

        return actual;
    }

    // Encuentra y devuelve el nodo según la posición
    private Nodo obtenerNodoPorPosicion(int posicion) {
        Nodo actual = inicioTablero;
        while (actual != null) {
            if (actual.posicion == posicion) {
                return actual;
            }
            actual = actual.siguiente;
        }
        return null;
    }

    // Lanza el dado y genera un número entre 1 y 6
    private int lanzarDado() {
        return dado.nextInt(6) + 1;
    }

    public static void main(String[] args) {
        SerpientesEscaleras juego = new SerpientesEscaleras();
        juego.iniciarJuego();
    }
}
